```python
from app.extensions import db
from werkzeug.security import generate_password_hash, check_password_hash

class User(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(80), unique=True, nullable=False)
    email = db.Column(db.String(120), unique=True, nullable=False)
    password_hash = db.Column(db.String(128))
    role = db.Column(db.String(20), default='user')
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    
    def set_password(self, password):
        self.password_hash = generate_password_hash(password)
    
    def check_password(self, password):
        return check_password_hash(self.password_hash, password)
```

## 4. 服务层实现

### 4.1 文章服务
```python
# app/services/article_service.py
class ArticleService:
    def create_article(self, title, content, author_id):
        article = Article(
            title=title,
            content=content,
            author_id=author_id
        )
        db.session.add(article)
        db.session.commit()
        return article
    
    def publish_article(self, article_id, platforms):
        article = Article.query.get_or_404(article_id)
        results = []
        
        for platform in platforms:
            try:
                # 获取平台账号
                account = PlatformAccount.query.filter_by(
                    user_id=article.author_id,
                    platform=platform
                ).first()
                
                if not account:
                    raise ValueError(f"No account for platform {platform}")
                
                # 发布文章
                result = self._publish_to_platform(
                    article,
                    platform,
                    account
                )
                results.append(result)
                
            except Exception as e:
                results.append({
                    'platform': platform,
                    'status': 'failed',
                    'error': str(e)
                })
        
        return results
```

### 4.2 平台服务
```python
# app/services/platform_service.py
class PlatformService:
    def bind_account(self, user_id, platform, auth_code):
        # 获取平台 access token
        token_info = self._get_platform_token(platform, auth_code)
        
        account = PlatformAccount(
            user_id=user_id,
            platform=platform,
            access_token=token_info['access_token'],
            refresh_token=token_info.get('refresh_token'),
            expires_at=datetime.utcnow() + timedelta(
                seconds=token_info['expires_in']
            )
        )
        
        db.session.add(account)
        db.session.commit()
        return account
```

```python
from celery import Celery

celery = Celery(
    'genflow',
    broker='redis://localhost:6379/0',
    backend='redis://localhost:6379/1'
)

celery.conf.update(
    task_serializer='json',
    accept_content=['json'],
    result_serializer='json',
    timezone='Asia/Shanghai',
    enable_utc=True
)
```

### 5.2 发布任务
```python
# app/tasks/article_tasks.py
@celery.task(bind=True, max_retries=3)
def publish_article_task(self, article_id, platforms):
    try:
        article_service = ArticleService()
        results = article_service.publish_article(
            article_id,
            platforms
        )
        return results
    except Exception as e:
        self.retry(exc=e, countdown=60)  # 1分钟后重试
```

## 6. 安全措施

### 6.1 认证中间件
```python
# app/utils/security.py
from functools import wraps
from flask import request, g
import jwt

def auth_required(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        token = request.headers.get('Authorization')
        if not token:
            return {'message': 'Missing token'}, 401
        
        try:
            data = jwt.decode(
                token,
                current_app.config['SECRET_KEY'],
                algorithms=['HS256']
            )
            g.user_id = data['user_id']
        except:
            return {'message': 'Invalid token'}, 401
            
        return f(*args, **kwargs)
    return decorated
```
